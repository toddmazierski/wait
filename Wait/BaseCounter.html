def initialize(total)
  # Attempt to prevent causing an infinite loop by being very strict about
  # the value passed.
  unless total.is_a?(Fixnum) and total > 0
    raise(ArgumentError, "invalid number of attempts: #{total.inspect}")
  end

  @total = total
end
